% ============================================================
% Packages
% ============================================================

% AMS theorem environments
\usepackage{amsthm}

% Math font
\bisusetup{
  math-font = xits,
}

% ============================================================
% Page Layout — BISU 2026 Requirements
% ============================================================

% Margins: 1 inch on all sides (2.54 cm)
\geometry{
  top    = 1in,
  bottom = 1in,
  left   = 1in,
  right  = 1in,
  headheight = 0pt,
  headsep    = 0pt,
}

% Line spacing: 1.5
\usepackage{setspace}
\onehalfspacing

% ============================================================
% Font Settings — Times New Roman, 12pt body
% ============================================================
% The bisuthesis class already loads Times New Roman via fontset.
% Ensure body text is 12pt (set in documentclass options via the class).
% Caption font size: 10-11pt

\usepackage{caption}
\captionsetup{
  font       = {small, singlespacing},
  labelfont  = {bf},
  justification = centering,
}

% ============================================================
% Page Numbers — Bottom center, 12pt Times New Roman
% ============================================================

\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\renewcommand\headrulewidth{0pt}
\fancyfoot[C]{\fontsize{12pt}{14pt}\selectfont\thepage}
\fancypagestyle{plain}{%
  \fancyhf{}
  \renewcommand\headrulewidth{0pt}
  \fancyfoot[C]{\fontsize{12pt}{14pt}\selectfont\thepage}
}
